    ##############################################

    @property
    def path(self):
        return self._path

    ##############################################

    def _parse(self):

        with open(str(self._path), 'rb') as f:
            source = f.read()

        return etree.fromstring(source)

    ##############################################

    @staticmethod
    def _get_xpath_element(root, path):

        return root.xpath(path)[0]
